vi z_func(string s)
{
    vi z(s.sz, 0);

    int l = 0;
    int r = 0;

    for (int i = 1; i < s.sz; i++)
    {
        int count = 0;
        if (i <= r)  count = min(z[i - l], r - i + 1);

        for (int j = count; i + j < s.sz && s[j] == s[i + j]; j++)
            count++;

        z[i] = count;

        if (i + count >= r)
        {
            l = i;
            r = i + count - 1;
        }

    }
    return z;
}
